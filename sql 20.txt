Select * from Users;
truncate table users;

Create Procedure EditUser 
@id int 
as
begin
Select * from Users Where Id = @id
End

Create Procedure UpdateUser
@id int, @Salutation int, @FirstName varchar(20), @LastName varchar(20), @Gender int, @Comments Varchar(50)
as
begin
Update Users Set Salutation= @Salutation, FirstName = @FirstName, LastName = @LastName, Gender = @Gender,Comments = @Comments From Users where ID = @id
End
